### State Management Complexity
```javascript
// Issue: Prop drilling through 6 levels
// Solution: Context + custom hooks

const UserContext = createContext();

export const useUser = () => {
  const context = useContext(UserContext);
  if (!context) throw new Error('Must use within provider');
  return context;
};

// Alternative: Zustand for simpler cases
import { create } from 'zustand';
const useStore = create(set => ({
  user: null,
  setUser: user => set({ user })
}));
```

### Component Design
```javascript
// âœ… Good: Composable, prop-based
function Card({ variant = 'default', size = 'md', children, ...props }) {
  return (
    <div className={`card card--${variant} card--${size}`} {...props}>
      {children}
    </div>
  );
}

// Usage flexibility
<Card><p>Content</p></Card>
<Card variant="elevated" size="lg"><p>Larger</p></Card>
```

### State Design
```javascript
// âœ… Good: Normalized, derived state
const [items, setItems] = useState([]);
const [selectedIds, setSelectedIds] = useState(new Set());

// Derive when needed
const selectedItems = useMemo(
  () => items.filter(item => selectedIds.has(item.id)),
  [items, selectedIds]
);

// âŒ Bad: Duplicate state
const [items, setItems] = useState([]);
const [selectedItems, setSelectedItems] = useState([]);
// Now you have to keep both in sync!
```

### Performance
```javascript
// âœ… Good: Strategic memoization
const MemoizedList = memo(List, (prev, next) =>
  prev.itemsCount === next.itemsCount &&
  prev.sortOrder === next.sortOrder
);

// âœ… Good: Code splitting
const Modal = lazy(() => import('./Modal'));
<Suspense fallback={<Spinner />}>
  <Modal />
</Suspense>

// âŒ Bad: Over-memoization
```
